## Task 1: Write the test (RED)

**File:** `packages/coding-agent/test/unit/audit-log-tool-loop.test.ts`

```ts
import { describe, expect, it } from 'vitest';
import { runToolLoop } from '../../src/agent/tool-loop.js';
import { AuditLog } from '../../src/observability/audit-log.js';
import { ToolRegistry } from '../../src/tools/registry.js';
import type { Tool, ToolResult, ChatMessage, ChatResult, LLMClient, ModelId, ToolCall } from '@deepwhale/llm';

// ---- minimal LLM mock (echo tool then stop) ----
function scriptedClient(responses: ReadonlyArray<ChatResult>): LLMClient {
  let idx = 0;
  return {
    model: 'mock' as ModelId,
    chat: async (): Promise<ChatResult> => {
      const r = responses[idx];
      if (!r) throw new Error(`scriptedClient: no response at index ${idx}`);
      idx += 1;
      return r;
    },
  } as LLMClient;
}

function okResult(content: string, toolCalls: ReadonlyArray<ToolCall> = []): ChatResult {
  const r: ChatResult = {
    model: 'mock' as ModelId,
    content,
    finish_reason: toolCalls.length > 0 ? 'tool_calls' : 'stop',
  };
  if (toolCalls.length > 0) r.tool_calls = toolCalls;
  return r;
}

function echoTool(): Tool {
  return {
    name: 'echo',
    description: 'returns args.text',
    risk: 'low',
    schema: { type: 'object', properties: { text: { type: 'string' } }, required: ['text'] },
    execute: async (args: { text: string }): Promise<ToolResult> => ({
      success: true,
      content: `echo: ${args.text}`,
    }),
  };
}

describe('AuditLog tool-loop integration (D-88 v5.0 observability+auditability)', () => {
  it('captures tool-call, tool-result, and loop-end events from runToolLoop (D-88 v5.0 integration)', async () => {
    // 2-turn script: 1 tool call (echo 'hi') + 1 final text stop.
    const client = scriptedClient([
      okResult('', [
        { id: 'tc-1', name: 'echo', args: { text: 'hi' } },
      ]),
      okResult('done'),
    ]);
    const registry = new ToolRegistry([echoTool()]);
    const auditLog = new AuditLog();
    const messages: ChatMessage[] = [
      { role: 'user', content: 'say hi' },
    ];

    await runToolLoop(client, messages, { registry, auditLog });

    // Audit log captured 3 events: tool-call, tool-result, loop-end.
    // Order is significant: tool-call BEFORE tool-result, both BEFORE loop-end.
    const events = auditLog.getEvents();
    expect(events).toHaveLength(3);

    expect(events[0]).toMatchObject({
      kind: 'tool-call',
      payload: { name: 'echo' },
    });
    expect(events[1]).toMatchObject({
      kind: 'tool-result',
      payload: { name: 'echo', ok: true },
    });
    expect(events[2]).toMatchObject({
      kind: 'loop-end',
      payload: { steps: 1 },
    });

    // Timestamps are monotonically non-decreasing (deterministic clock defaults
    // to Date.now; for the seed test we just assert non-decreasing).
    expect(events[1]!.timestamp).toBeGreaterThanOrEqual(events[0]!.timestamp);
    expect(events[2]!.timestamp).toBeGreaterThanOrEqual(events[1]!.timestamp);
  });
});
```

Run: `./node_modules/.bin/vitest.cmd run packages/coding-agent/test/unit/audit-log-tool-loop.test.ts --reporter=verbose`

Expected: 1 test fails because `ToolLoopOptions` doesn't yet have an `auditLog` field. RED confirmed.
