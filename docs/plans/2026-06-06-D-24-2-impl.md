## 3. store (nanostores)

`packages/tui-ink/src/store/ui.ts`:

```ts
import { atom, map } from 'nanostores'

export type Mode = 'idle' | 'streaming' | 'confirm' | 'tool' | 'error'

export interface UiState {
  mode: Mode
  usage: string | null    // formatUsageStatus output
  model: string           // model display name
  pendingConfirm: { prompt: string; toolName: string } | null
  // transcript / history 由 React useState 维护 (Ink <Static> 期望数组 prop)
}

export const $uiState = map<UiState>({
  mode: 'idle',
  usage: null,
  model: '',
  pendingConfirm: null,
})

// 累积消息 (user / assistant / tool)
export const $transcript = atom<TranscriptEntry[]>([])

export interface TranscriptEntry {
  kind: 'user' | 'assistant' | 'tool'
  text: string
  status?: 'success' | 'error'
  toolName?: string
  durationMs?: number
}
```
