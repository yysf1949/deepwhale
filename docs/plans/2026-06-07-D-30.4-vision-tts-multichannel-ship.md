```ts
import { defineTool } from '../tool-helper.js';
import { promises as fs } from 'node:fs';
import { runOneTurn } from '../llm-runner.js';

export const visionAnalyze = defineTool<{ source: string; prompt?: string }, { success: boolean; content: string }>({
  name: 'vision_analyze',
  description: 'Analyze an image (local path or URL) using a vision-capable LLM',
  parameters: {
    type: 'object',
    properties: {
      source: { type: 'string', description: 'Image path or URL' },
      prompt: { type: 'string', description: 'Analysis prompt' },
    },
    required: ['source'],
  },
  async execute(args) {
    try {
      let dataUrl = args.source;
      if (!args.source.startsWith('http')) {
        const buf = await fs.readFile(args.source);
        const base64 = buf.toString('base64');
        const ext = args.source.split('.').pop()?.toLowerCase();
        const mime = ext === 'png' ? 'image/png' : 'image/jpeg';
        dataUrl = `data:${mime};base64,${base64}`;
      }
      const turn = await runOneTurn(/* vision client */ {} as any, [
        { role: 'user', content: args.prompt || 'describe this image' },
        { role: 'system', content: `image: ${dataUrl}` },
      ], [], {});
      return { success: true, content: turn.kind === 'chat' ? turn.assistant : '(no result)' };
    } catch (e) {
      return { success: false, content: `vision error: ${e instanceof Error ? e.message : String(e)}` };
    }
  },
});
```

```ts
export interface TelegramConfig { token: string; chatId: number }

export class TelegramChannel {
  private polling = false;
  constructor(private config: TelegramConfig, private onMessage: (text: string) => Promise<string>) {}

  async start(): Promise<void> {
    this.polling = true;
    let offset = 0;
    while (this.polling) {
      try {
        const res = await fetch(`https://api.telegram.org/bot${this.config.token}/getUpdates?offset=${offset}&timeout=30`);
        const data = await res.json() as { result: Array<{ update_id: number; message: { chat: { id: number }; text: string } }> };
        for (const update of data.result) {
          offset = update.update_id + 1;
          if (update.message.text) {
            const reply = await this.onMessage(update.message.text);
            await this.sendMessage(update.message.chat.id, reply);
          }
        }
      } catch (e) {
        // log and retry
      }
    }
  }

  stop(): void { this.polling = false; }

  async sendMessage(chatId: number, text: string): Promise<void> {
    await fetch(`https://api.telegram.org/bot${this.config.token}/sendMessage`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ chat_id: chatId, text }),
    });
  }
}
```

```ts
export interface DiscordConfig { token: string; channelId: string }

export class DiscordChannel {
  private ws: WebSocket | null = null;
  constructor(private config: DiscordConfig, private onMessage: (text: string) => Promise<string>) {}

  async start(): Promise<void> {
    // Discord gateway: GET /gateway/bot, then connect via WS
    const res = await fetch('https://discord.com/api/v10/gateway/bot', {
      headers: { Authorization: `Bot ${this.config.token}` },
    });
    const { url } = await res.json() as { url: string };
    this.ws = new WebSocket(`${url}?v=10&encoding=json`);
    this.ws.onmessage = async (e) => {
      const data = JSON.parse(e.data);
      if (data.t === 'MESSAGE_CREATE' && data.d.channel_id === this.config.channelId) {
        const reply = await this.onMessage(data.d.content);
        await this.sendMessage(reply);
      }
    };
  }

  stop(): void { this.ws?.close(); }

  async sendMessage(text: string): Promise<void> {
    await fetch(`https://discord.com/api/v10/channels/${this.config.channelId}/messages`, {
      method: 'POST',
      headers: { Authorization: `Bot ${this.config.token}`, 'Content-Type': 'application/json' },
      body: JSON.stringify({ content: text }),
    });
  }
}
```

```ts
export interface ChannelMessage { channel: 'telegram' | 'discord' | 'tui'; text: string; userId?: string }

export class ChannelRouter {
  private handlers: Array<(m: ChannelMessage) => Promise<string>> = [];

  onMessage(h: (m: ChannelMessage) => Promise<string>): void { this.handlers.push(h); }

  async dispatch(msg: ChannelMessage): Promise<string> {
    let last = '';
    for (const h of this.handlers) last = await h(msg);
    return last;
  }
}
```
