```ts
import type { Capability, CapabilitySource, RiskLevel } from './capability.js';
import type { ToolRegistryProfile } from '../tools/registry.js';

export interface RegisterCapabilityInput {
  id: string;
  source: CapabilitySource;
  riskLevel: RiskLevel;
  profiles: ReadonlyArray<ToolRegistryProfile>;
  description?: string;
  sideEffects?: ReadonlyArray<string>;
}

export class CapabilityRegistry {
  private capabilities = new Map<string, Capability>();

  register(input: RegisterCapabilityInput): void {
    if (this.capabilities.has(input.id)) {
      throw new Error(`duplicate capability id: ${input.id}`);
    }
    this.capabilities.set(input.id, Object.freeze({ ...input }));
  }

  get(id: string): Capability | undefined {
    return this.capabilities.get(id);
  }

  list(filter?: { profiles?: ReadonlyArray<ToolRegistryProfile> }): ReadonlyArray<Capability> {
    const all = Array.from(this.capabilities.values());
    if (!filter?.profiles) return all;
    const wanted = new Set<ToolRegistryProfile>(filter.profiles);
    return all.filter((c) => c.profiles.some((p) => wanted.has(p)));
  }

  size(): number {
    return this.capabilities.size;
  }
}

export function createCapabilityRegistry(): CapabilityRegistry {
  return new CapabilityRegistry();
}
```

**Step 3: Implement `loadSkill` as a free function**

Add to `packages/coding-agent/src/util/skill-loader.ts`:

```ts
export interface LoadSkillOptions {
  root: string;
  availableCapabilities: ReadonlyArray<string>;
}

export interface LoadedSkillWithMeta {
  name: string;
  content: string;
  capabilities: ReadonlyArray<string>;
}

/** Parse SKILL.md frontmatter (YAML) and validate capabilities. */
export async function loadSkill(options: LoadSkillOptions): Promise<LoadedSkillWithMeta> {
  const skillFile = join(options.root, 'SKILL.md');
  const raw = await fs.readFile(skillFile, 'utf8');
  // Parse frontmatter between leading `---` lines
  const fm = /^---\n([\s\S]*?)\n---\n?/.exec(raw);
  if (!fm) throw new Error(`SKILL.md missing frontmatter: ${skillFile}`);
  const fmBody = fm[1];
  const nameMatch = /^name:\s*(\S+)/m.exec(fmBody);
  if (!nameMatch) throw new Error(`SKILL.md missing 'name' in frontmatter: ${skillFile}`);
  const name = nameMatch[1];
  const capMatch = /capabilities:\s*\n((?:\s*-\s*.+\n?)*)/.exec(fmBody);
  const capabilities = capMatch
    ? capMatch[1].split('\n').map((l) => l.replace(/^\s*-\s*/, '').trim()).filter(Boolean)
    : [];
  // Validate
  for (const cap of capabilities) {
    if (!options.availableCapabilities.includes(cap)) {
      throw new Error(`missing capability: ${cap} (skill ${name})`);
    }
  }
  return { name, content: raw, capabilities };
}
```

```ts
export interface PlanGate1Shape {
  repoRoot: string;
  loc: number;
  supportedFiles: number;
  symbols: number;
  references: number;
  callEdges: number;
  elapsedMs: number;
  entry: { file: string; symbol: string; line: number; col: number; kind: string };
  callChain: ReadonlyArray<{ caller: string; callee: string; line: number; file: string }>;
  modificationPoint: { file: string; symbol: string; line: number; col: number; kind: string };
  passed: boolean;
}

export function toPlanShape(result: Gate1Result): PlanGate1Shape {
  return {
    repoRoot: result.repoPath,
    loc: result.metrics.loc,
    supportedFiles: result.metrics.supportedFiles,
    symbols: result.metrics.symbolsIndexed,
    references: result.metrics.referencesIndexed,
    callEdges: result.metrics.callEdges,
    elapsedMs: result.metrics.elapsedMs,
    entry: result.evidence.entry,
    callChain: result.evidence.callChain,
    modificationPoint: result.evidence.modificationPoint,
    passed: result.passed,
  };
}
```
