// packages/coding-agent/src/tools/delegate-task.ts
import { defineTool } from '../tool-helper.js';
import { runOneTurn } from '../llm-runner.js';

export interface SubTask { prompt: string; model?: string }

export const delegateTask = defineTool<{ tasks: SubTask[]; concurrency?: number }, { success: boolean; content: string }>({
  name: 'delegate_task',
  description: 'Run sub-tasks in parallel subagents (max concurrency 5)',
  parameters: {
    type: 'object',
    properties: {
      tasks: { type: 'array', items: { type: 'object' } },
      concurrency: { type: 'number' },
    },
    required: ['tasks'],
  },
  async execute(args) {
    const limit = Math.min(args.concurrency ?? 3, 5);
    const results: string[] = [];
    const chunks: SubTask[][] = [];
    for (let i = 0; i < args.tasks.length; i += limit) {
      chunks.push(args.tasks.slice(i, i + limit));
    }
    for (const chunk of chunks) {
      const chunkResults = await Promise.all(
        chunk.map(async (t, i) => {
          try {
            const turn = await runOneTurn(/* mock client */ { model: 'mock' } as any, t.prompt, [], {});
            return `[task ${i}] ok: ${turn.kind}`;
          } catch (e) {
            return `[task ${i}] error: ${e instanceof Error ? e.message : String(e)}`;
          }
        })
      );
      results.push(...chunkResults);
    }
    return { success: true, content: results.join('\n') };
  },
});

// Modify session-index.ts to use sql.js
import initSqlJs from 'sql.js';
import { promises as fs } from 'node:fs';
import { join } from 'node:path';

let SQL: any;
async function getSQL() {
  if (!SQL) SQL = await initSqlJs();
  return SQL;
}

export class SessionIndex {
  private db: any;
  constructor(private rootDir: string) {}

  async init() {
    const sql = await getSQL();
    const dbPath = join(this.rootDir, 'sessions.db');
    try {
      const buf = await fs.readFile(dbPath);
      this.db = new sql.Database(new Uint8Array(buf));
    } catch {
      this.db = new sql.Database();
      this.db.exec(`CREATE VIRTUAL TABLE IF NOT EXISTS sessions USING fts5(id, path, first_user, content)`);
    }
  }

  async index(entry: SessionEntry, content: string) {
    if (!this.db) await this.init();
    this.db.run(`INSERT INTO sessions (id, path, first_user, content) VALUES (?, ?, ?, ?)`,
      [entry.id, entry.path, entry.firstUser, content]);
    await this.persist();
  }

  async search(query: string): Promise<SessionEntry[]> {
    if (!this.db) await this.init();
    const stmt = this.db.prepare(`SELECT id, path, first_user as firstUser FROM sessions WHERE content MATCH ?`);
    stmt.bind([query]);
    const results: SessionEntry[] = [];
    while (stmt.step()) {
      results.push({ ...stmt.getAsObject(), messageCount: 0, createdAt: 0 });
    }
    return results;
  }

  private async persist() {
    const data = this.db.export();
    await fs.writeFile(join(this.rootDir, 'sessions.db'), Buffer.from(data));
  }
}
```

import { promises as fs } from 'node:fs';
import { join } from 'node:path';

export class SkillLoader {
  constructor(private rootDir: string) {}

  async listSkills(): Promise<Array<{ name: string; content: string }>> {
    const skillsDir = join(this.rootDir, 'skills');
    try {
      const entries = await fs.readdir(skillsDir, { withFileTypes: true });
      const skills = [];
      for (const e of entries) {
        if (e.isDirectory()) {
          const content = await fs.readFile(join(skillsDir, e.name, 'SKILL.md'), 'utf8').catch(() => '');
          skills.push({ name: e.name, content });
        }
      }
      return skills;
    } catch (e) {
      if ((e as NodeJS.ErrnoException).code === 'ENOENT') return [];
      throw e;
    }
  }
}

import { CronStore, type CronJob } from './cron-store.js';

export class CronDaemon {
  private timer: NodeJS.Timeout | null = null;
  constructor(private store: CronStore, private onTick: (job: CronJob) => Promise<void>) {}

  start(intervalMs = 60_000): void {
    this.timer = setInterval(async () => {
      const jobs = await this.store.list();
      for (const job of jobs.filter((j) => j.enabled)) {
        await this.onTick(job);
      }
    }, intervalMs);
    this.timer.unref?.();
  }

  stop(): void {
    if (this.timer) { clearTimeout(this.timer); this.timer = null; }
  }
}
```
