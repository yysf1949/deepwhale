// packages/coding-agent/src/tools/read-file.ts
import { defineTool } from '../tool-helper.js';
import { promises as fs } from 'node:fs';

export const readFile = defineTool<{ path: string; offset?: number; limit?: number }, { success: boolean; content: string }>({
  name: 'read_file',
  description: 'Read file contents (optionally with offset/limit)',
  parameters: {
    type: 'object',
    properties: {
      path: { type: 'string', description: 'File path' },
      offset: { type: 'number', description: 'Start line (0-based)' },
      limit: { type: 'number', description: 'Max lines' },
    },
    required: ['path'],
  },
  async execute(args) {
    try {
      const content = await fs.readFile(args.path, 'utf8');
      if (args.offset === undefined && args.limit === undefined) {
        return { success: true, content };
      }
      const lines = content.split('\n');
      const start = args.offset ?? 0;
      const end = args.limit ? start + args.limit : lines.length;
      return { success: true, content: lines.slice(start, end).join('\n') };
    } catch (e) {
      return { success: false, content: `read error: ${e instanceof Error ? e.message : String(e)}` };
    }
  },
});
```

// packages/coding-agent/src/tools/patch.ts
import { defineTool } from '../tool-helper.js';
import { promises as fs } from 'node:fs';

export const patch = defineTool<{ path: string; oldString: string; newString: string }, { success: boolean; content: string }>({
  name: 'patch',
  description: 'Find and replace a unique string in a file',
  parameters: {
    type: 'object',
    properties: {
      path: { type: 'string' },
      oldString: { type: 'string' },
      newString: { type: 'string' },
    },
    required: ['path', 'oldString', 'newString'],
  },
  async execute(args) {
    try {
      const content = await fs.readFile(args.path, 'utf8');
      const occurrences = content.split(args.oldString).length - 1;
      if (occurrences === 0) return { success: false, content: 'oldString not found' };
      if (occurrences > 1) return { success: false, content: `oldString matches ${occurrences} times (must be unique)` };
      const newContent = content.replace(args.oldString, args.newString);
      await fs.writeFile(args.path, newContent);
      return { success: true, content: 'patched' };
    } catch (e) {
      return { success: false, content: `patch error: ${e instanceof Error ? e.message : String(e)}` };
    }
  },
});
```

// packages/coding-agent/src/tools/search-files.ts
import { defineTool } from '../tool-helper.js';
import { execFileSync } from 'node:child_process';

export const searchFiles = defineTool<{ pattern: string; path: string; glob?: string }, { success: boolean; content: string }>({
  name: 'search_files',
  description: 'Search file contents using ripgrep',
  parameters: {
    type: 'object',
    properties: {
      pattern: { type: 'string', description: 'Search pattern' },
      path: { type: 'string', description: 'Search root' },
      glob: { type: 'string', description: 'File glob (e.g. *.ts)' },
    },
    required: ['pattern', 'path'],
  },
  async execute(args) {
    try {
      const args_: string[] = ['--color=never', '-n', '--no-heading', args.pattern, args.path];
      if (args.glob) args_.push('--glob', args.glob);
      const output = execFileSync('rg', args_, { encoding: 'utf8', maxBuffer: 1024 * 1024 });
      return { success: true, content: output || '(no matches)' };
    } catch (e: any) {
      if (e.status === 1) return { success: true, content: '(no matches)' };
      return { success: false, content: `search error: ${e.message}` };
    }
  },
});
```

// packages/coding-agent/src/tools/execute-code.ts
import { defineTool } from '../tool-helper.js';
import { spawn } from 'node:child_process';
import { writeFile, mkdtemp, rm } from 'node:fs/promises';
import { tmpdir } from 'node:os';
import { join } from 'node:path';

export const executeCode = defineTool<{ language: 'python' | 'javascript'; code: string }, { success: boolean; content: string }>({
  name: 'execute_code',
  description: 'Execute code in a subprocess with timeout (Python or JavaScript)',
  parameters: {
    type: 'object',
    properties: {
      language: { type: 'string', enum: ['python', 'javascript'] },
      code: { type: 'string' },
    },
    required: ['language', 'code'],
  },
  async execute(args) {
    const dir = await mkdtemp(join(tmpdir(), 'dw-exec-'));
    try {
      const ext = args.language === 'python' ? 'py' : 'js';
      const file = join(dir, `exec.${ext}`);
      await writeFile(file, args.code);
      const cmd = args.language === 'python' ? 'python3' : 'node';
      return new Promise((resolve) => {
        const child = spawn(cmd, [file], { cwd: dir });
        let stdout = '';
        let stderr = '';
        child.stdout.on('data', (d) => { stdout += d.toString(); });
        child.stderr.on('data', (d) => { stderr += d.toString(); });
        const timer = setTimeout(() => {
          child.kill('SIGTERM');
          resolve({ success: false, content: `timeout (30s)\nstdout: ${stdout}\nstderr: ${stderr}` });
        }, 30_000);
        child.on('close', (code) => {
          clearTimeout(timer);
          const ok = code === 0;
          resolve({ success: ok, content: ok ? stdout : `exit ${code}\nstdout: ${stdout}\nstderr: ${stderr}` });
        });
      });
    } finally {
      await rm(dir, { recursive: true, force: true });
    }
  },
});
```

// packages/coding-agent/src/tools/todo.ts
import { defineTool } from '../tool-helper.js';
import { promises as fs } from 'node:fs';
import { join } from 'node:path';

export interface TodoItem { id: string; text: string; done: boolean }

export class TodoStore {
  constructor(private rootDir: string) {}
  private get todoPath(): string { return join(this.rootDir, 'todos', 'current.json'); }

  async list(): Promise<TodoItem[]> {
    try {
      const data = await fs.readFile(this.todoPath, 'utf8');
      return JSON.parse(data) as TodoItem[];
    } catch (e) {
      if ((e as NodeJS.ErrnoException).code === 'ENOENT') return [];
      throw e;
    }
  }

  async add(text: string): Promise<TodoItem> {
    const items = await this.list();
    const item: TodoItem = { id: String(items.length + 1), text, done: false };
    items.push(item);
    await this.save(items);
    return item;
  }

  async markDone(id: string): Promise<void> {
    const items = await this.list();
    const item = items.find((i) => i.id === id);
    if (item) { item.done = true; await this.save(items); }
  }

  private async save(items: TodoItem[]): Promise<void> {
    await fs.mkdir(join(this.rootDir, 'todos'), { recursive: true });
    await fs.writeFile(this.todoPath, JSON.stringify(items, null, 2));
  }
}

export const todo = defineTool<{ action: 'list' | 'add' | 'mark_done'; text?: string; id?: string }, { success: boolean; content: string }>({
  name: 'todo',
  description: 'Manage todos (list/add/mark_done)',

      text: { type: 'string' },
      id: { type: 'string' },
    },
    required: ['action'],
  },

    // Implementation uses TodoStore

  },
});

import { Box, Text } from 'ink';
import { useStore } from '@nanostores/react';
import { $todos, type TodoItem } from '../store/ui.js';
import { colorize, type TuiTheme, THEMES } from '../theme/index.js';
import type { ReactElement } from 'react';

export function TodoList({ theme = THEMES.default }: { theme?: TuiTheme }): ReactElement {

  return (
    <Box flexDirection="column">
      <Text color={theme.header}>Todos:</Text>

        <Text key={item.id}>
          {colorize(item.done ? '☑' : '☐', item.done ? 'success' : 'prompt', theme)}
          {' '}
          {item.text}
        </Text>
      ))}
    </Box>
  );
}
