```ts
describe('browser visual element labels', () => {
  it('returns labeled visible elements with confidence', () => {
    const obs = observeHtml({
      url: 'https://example.com',
      title: 'Example',

    const button = obs.visibleElements.find((e) => e.type === 'button');
    const link = obs.visibleElements.find((e) => e.type === 'link');
```

## Task A.2: Strategic Recovery

**Files:**
- Modify: `packages/coding-agent/src/browser/planner.ts` (add `failureHistory` parameter to `planBrowserAction`)
- Test: `packages/coding-agent/test/unit/browser-recovery.test.ts` (new)

**Step 1: Test**

```ts
import { describe, expect, it } from 'vitest';
import { planBrowserAction } from '../../src/browser/planner.js';

describe('browser strategic recovery', () => {
  it('returns a different action when the previous click on a button failed', () => {
    const obs = {
      url: 'https://example.com',
      title: 'Example',
      domSummary: '<main>: 1 button, 1 input',
      visibleElements: [
        { type: 'button' as const, text: 'Submit', selector: 'button.submit' },
        { type: 'input' as const, text: 'Email', selector: 'input#email' },
      ],
      actionHistory: [],
    };
    const recovery = planBrowserAction({
      userIntent: 'submit the form',
      observation: obs,
      failureHistory: [
        { action: { type: 'click', target: 'button.submit' }, failureReason: 'element-not-clickable' },
      ],
    });
    expect(recovery.type).not.toBe('click');
    expect(['type', 'navigate', 'skip']).toContain(recovery.type);
  });

  it('returns a skip action when the requested action recently failed', () => {
    const obs = {
      url: 'https://example.com',
      title: 'Example',
      domSummary: '<main>: 1 button',
      visibleElements: [{ type: 'button' as const, text: 'X', selector: 'button.x' }],
      actionHistory: [],
    };
    const result = planBrowserAction({
      userIntent: 'click the X button',
      observation: obs,
      failureHistory: [
        { action: { type: 'click', target: 'button.x' }, failureReason: 'recently-failed' },
      ],
    });
    expect(result.type).toBe('skip');
  });
});
```

**Step 2: Run — confirm fail**

**Step 3: Implement** in `planner.ts`:
- Add `failureHistory?: Array<{ action: { type: string; target: string }; failureReason: string }>` parameter.
- If the new action matches a previous failure, return `{ type: 'skip', target: null, reason: 'recently-failed' }`.
- For recovery: if the previous action was a `click` and it failed, suggest a `type` action on an adjacent `input` element if available; otherwise `navigate`.

**Step 4: Run — confirm pass**
Expected: 2/2 pass.

**Step 5: Commit**

```ts
        { action: { type: 'click', target: 'button.broken' }, failureReason: 'timeout' },
```
