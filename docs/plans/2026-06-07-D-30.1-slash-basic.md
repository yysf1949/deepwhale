**Step 4.3: Implement /status**
```ts
if (line === '/status') {
  if (ctx.getStatus) {
    const s = ctx.getStatus();
    const status = [
      'Current status:',
      `  model:      ${s.model}`,
      `  session:    ${s.sessionPath ?? '(no session)'}`,
      `  ema:        ${s.emaSampleCount} samples`,
      `  theme:      ${s.theme}`,
      `  uptime:     ${Math.floor(s.uptimeMs / 1000)}s`,
    ].join('\n');
    ctx.out.write(`${status}\n\n`);
  } else {
    ctx.out.write('status: (no status provider wired)\n\n');
  }
  ctx.prompt();
  return { handled: true };
}
```

```ts

}
```
