### Commit 4: `test(tui-ink): Confirm ink-testing-library 单测 (D-29.3.4)`

**测试设计** (3-4 it, 跟 ReplConfirmController 状态联动):

```tsx
import { describe, it, expect } from 'vitest';
import { render } from 'ink-testing-library';
import { atom } from 'nanostores';
import { Confirm } from '../../src/components/Confirm.js';
import { $uiState } from '../../src/store/ui.js';

describe('tui-ink/components/Confirm', () => {
  beforeEach(() => {
    $uiState.set({ ...$uiState.get(), pendingConfirm: null });
  });

  it('无 pendingConfirm 时不渲染 (conditional)', () => {
    const { lastFrame } = render(<Confirm />);
    expect(lastFrame()).not.toContain('Allow');
    expect(lastFrame()).not.toContain('[y/N]');
  });

  it('有 pendingConfirm 渲染 "Allow <tool>? (<reason>) [y/N]: "', () => {
    $uiState.set({
      ...$uiState.get(),
      pendingConfirm: { tool: 'bash', reason: 'overwrite file foo.txt' },
    });
    const { lastFrame } = render(<Confirm />);
    expect(lastFrame()).toContain('Allow bash?');
    expect(lastFrame()).toContain('overwrite file foo.txt');
    expect(lastFrame()).toContain('[y/N]');
  });

  it('tool 名超长截断 (防御 X 字符注入)', () => {
    $uiState.set({
      ...$uiState.get(),
      pendingConfirm: { tool: 'a'.repeat(200), reason: 'test' },
    });
    const { lastFrame } = render(<Confirm />);
    const frame = lastFrame()!;
    expect(frame).toContain('Allow aaaa');
    expect(frame.length).toBeLessThan(200);
  });
});
```
