```ts
export type ReviewStatus = 'approve' | 'request_changes';

export interface ReviewInput {
  commands: ReadonlyArray<string>;
}

export interface ReviewResult {
  status: ReviewStatus;
  details?: Array<{ command: string; exitCode: number; stdout: string; stderr: string }>;
}

export interface CommandRunner {
  runCommand(command: string): Promise<{ exitCode: number; stdout: string; stderr: string }>;
}

export interface Reviewer {
  review(input: ReviewInput): Promise<ReviewResult>;
  writeFile(path: string, content: string): Promise<void>;
}

export function createReviewer(opts: { runCommand: CommandRunner['runCommand'] }): Reviewer {
  return {
    async review({ commands }) {
      const details = [];
      let allPassed = true;
      for (const cmd of commands) {
        const result = await opts.runCommand(cmd);
        details.push({ command: cmd, ...result });
        if (result.exitCode !== 0) allPassed = false;
      }
      return { status: allPassed ? 'approve' : 'request_changes', details };
    },
    async writeFile(_path, _content) {
      throw new Error('reviewer cannot modify files');
    },
  };
}
```

```ts
export interface Gate2Step {
  index: number;
  tool: string;
  summary: string;
  retry: boolean;
}

export interface Gate2Transcript {
  goal: string;
  steps: ReadonlyArray<Gate2Step>;
}

export interface Gate2Evaluation {
  passed: boolean;
  toolCalls: number;
  retries: number;
  goalDriftDetected: boolean;
  reason?: string;
}

export function evaluateGate2Transcript(transcript: Gate2Transcript): Gate2Evaluation {
  const toolCalls = transcript.steps.length;
  const retries = transcript.steps.filter((s) => s.retry).length;
  // Goal drift heuristic: if the steps diverge in keyword vocabulary from the goal
  const goalTokens = new Set(transcript.goal.toLowerCase().split(/\W+/).filter((t) => t.length > 3));
  let drift = false;
  let driftReason: string | undefined;
  for (const step of transcript.steps) {
    const stepTokens = step.summary.toLowerCase().split(/\W+/).filter((t) => t.length > 3);
    const overlap = stepTokens.filter((t) => goalTokens.has(t)).length;
    if (stepTokens.length > 0 && overlap === 0) {
      drift = true;
      driftReason = 'goal-drift';
      break;
    }
  }
  const passed = toolCalls >= 30 && toolCalls <= 50 && !drift;
  return {
    passed,
    toolCalls,
    retries,
    goalDriftDetected: drift,
    ...(driftReason ? { reason: driftReason } : {}),
  };
}
```
